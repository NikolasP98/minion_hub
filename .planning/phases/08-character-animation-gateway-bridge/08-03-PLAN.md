Add a `renderNameLabels()` function in renderer.ts. Call it from `renderScene()` AFTER `renderBubbles()` (so labels draw on top of everything):

```typescript
function renderNameLabels(
  ctx: CanvasRenderingContext2D,
  characters: Character[],
  offsetX: number,
  offsetY: number,
  zoom: number,
): void {
  ctx.save();
  ctx.font = "bold 7px 'JetBrains Mono NF'";
  ctx.textAlign = 'center';
  ctx.textBaseline = 'bottom';

  for (const ch of characters) {
    // Resolve character name: charId -> instanceId -> agentId -> gw.agents name
    const instanceId = getInstanceForCharId(ch.id);
    if (!instanceId) continue;
    const agentInst = workshopState.agents[instanceId];
    if (!agentInst) continue;

    let name: string;
    if (ch.isSubagent) {
      // D-09: Sub-agent label format: "[parent name] >"
      const parentInstanceId = ch.parentAgentId !== null
        ? getInstanceForCharId(ch.parentAgentId) : undefined;
      const parentInst = parentInstanceId ? workshopState.agents[parentInstanceId] : undefined;
      const parentName = parentInst
        ? (gw.agents.find(a => a.id === parentInst.agentId)?.name ?? parentInst.agentId)
        : 'Agent';
      name = parentName.slice(0, 10) + ' >';
    } else {
      const agentName = gw.agents.find(a => a.id === agentInst.agentId)?.name ?? agentInst.agentId;
      // Truncate to 12 chars with ellipsis per UI-SPEC copywriting contract
      name = agentName.length > 12 ? agentName.slice(0, 12) + '...' : agentName;
    }

    const sittingOffset = ch.state === CharacterState.TYPE || (ch.seatId && ch.state === CharacterState.IDLE)
      ? CHARACTER_SITTING_OFFSET_PX : 0;
    const px = (ch.x + offsetX) * zoom;
    const py = (ch.y - 10 - sittingOffset + offsetY) * zoom;

    // Sub-agent labels at 0.7 opacity per UI-SPEC
    const alpha = ch.isSubagent ? 0.7 : 1.0;

    // Drop shadow
    ctx.globalAlpha = alpha * 0.8;
    ctx.fillStyle = 'rgba(0,0,0,0.8)';
    ctx.fillText(name, px + zoom, py + zoom);

    // Main text
    ctx.globalAlpha = alpha;
    ctx.fillStyle = '#ffffff';
    ctx.fillText(name, px, py);
  }

  ctx.restore();
}
```

```typescript
/**
 * Start dragging a character. Returns true if drag started.
 * Only one character can be dragged at a time.
 */
export function dragCharacter(
  id: number,
  characters: Map<number, Character>,
): boolean {
  if (draggedCharId !== null) return false; // already dragging
  const ch = characters.get(id);
  if (!ch) return false;
  draggedCharId = id;
  dragOriginalSeatId = ch.seatId;
  dragOriginalCol = ch.tileCol;
  dragOriginalRow = ch.tileRow;
  return true;
}

/**
 * Update dragged character position (world coordinates).
 */
export function updateDragPosition(worldX: number, worldY: number): void {
  dragWorldX = worldX;
  dragWorldY = worldY;
}

/**
 * Drop the dragged character. Returns the charId if drop was valid, null otherwise.
 */
export function dropCharacter(
  office: OfficeState,
  worldX: number,
  worldY: number,
): number | null {
  if (draggedCharId === null) return null;
  const charId = draggedCharId;
  const ch = office.characters.get(charId);

  // Find which tile the cursor is over
  const dropCol = Math.floor(worldX / TILE_SIZE);
  const dropRow = Math.floor(worldY / TILE_SIZE);

  // Find if there's a seat at this tile
  let targetSeatId: string | null = null;
  for (const [seatId, seat] of office.seats) {
    if (seat.seatCol === dropCol && seat.seatRow === dropRow && !seat.assigned) {
      targetSeatId = seatId;
      break;
    }
    // Also check if cursor is on the desk tile adjacent to an unoccupied seat
    // (seats face desks, so the desk is in the facing direction)
  }

  if (targetSeatId && ch) {
    // Valid drop on unoccupied seat
    office.reassignSeat(charId, targetSeatId);
    draggedCharId = null;
    return charId;
  }

  // Invalid drop — snap back to original seat
  cancelDrag(office);
  return null;
}

/**
 * Cancel drag — return character to original position.
 */
export function cancelDrag(office: OfficeState): void {
  if (draggedCharId === null) return;
  const ch = office.characters.get(draggedCharId);
  if (ch) {
    ch.tileCol = dragOriginalCol;
    ch.tileRow = dragOriginalRow;
    ch.x = dragOriginalCol * TILE_SIZE + TILE_SIZE / 2;
    ch.y = dragOriginalRow * TILE_SIZE + TILE_SIZE / 2;
  }
  draggedCharId = null;
}
```

```typescript
// Drag state
let isDragging = false;

function handleCanvasMousedown(e: MouseEvent) {
  if (e.button !== 0) return; // left click only
  const worldPos = screenToWorld(e.offsetX, e.offsetY);
  // Hit test: find character at click position
  const hitChar = findCharacterAtWorld(office, worldPos.x, worldPos.y);
  if (hitChar) {
    const started = dragCharacter(hitChar.id, office.characters);
    if (started) {
      isDragging = true;
      updateDragPosition(worldPos.x, worldPos.y);
      e.preventDefault();
    }
  }
}

function handleCanvasMousemove(e: MouseEvent) {
  if (!isDragging) return;
  const worldPos = screenToWorld(e.offsetX, e.offsetY);
  updateDragPosition(worldPos.x, worldPos.y);
}

function handleMouseup(e: MouseEvent) {
  if (!isDragging) return;
  isDragging = false;
  const worldPos = screenToWorld(e.offsetX ?? 0, e.offsetY ?? 0);
  dropCharacter(office, worldPos.x, worldPos.y);
}

function handleKeydown(e: KeyboardEvent) {
  if (e.key === 'Escape' && isDragging) {
    isDragging = false;
    cancelDrag(office);
  }
}
```
